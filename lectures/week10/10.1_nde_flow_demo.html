def make_synthetic_data(N, device=device):
    ''' complicated 4D distribution: 
    - (x0, x1): 3-component Gaussian mixture with different means & covariances
    - x2: nonlinear function of (x0, x1) + noise
          x2 = sin(x0) + 0.5 * x1^2 + eps
    - x3: heavy-tailed (Student-t-like) variable
    '''
    mus = torch.stack([
        torch.tensor([0.0, 0.0], device=device),
        torch.tensor([3.0, -3.0], device=device),
        torch.tensor([-3.0, 3.0], device=device)]) 

    covs = []
    covs.append(torch.tensor([[1.0, 0.2],
                              [0.2, 1.0]], device=device))
    covs.append(torch.tensor([[2.0, 1.5],
                              [1.5, 2.0]], device=device))
    covs.append(torch.tensor([[0.5, -0.3],
                              [-0.3, 0.5]], device=device))
    covs = torch.stack(covs, dim=0)  
    pis = torch.tensor([0.4, 0.35, 0.25], device=device)

    comp_idx = torch.multinomial(pis, num_samples=N, replacement=True)  # [N]

    # For each sample, get its mean and covariance
    mu = mus[comp_idx]           # [N, 2]
    cov = covs[comp_idx]         # [N, 2, 2]

    # Sample from N(0, I) and transform via Cholesky of cov
    eps = torch.randn(N, 2, device=device)  # [N, 2]
    # Cholesky factor per component
    L = torch.linalg.cholesky(cov)          # [N, 2, 2]
    xy = (L @ eps.unsqueeze(-1)).squeeze(-1) + mu  # [N, 2]

    x0 = xy[:, 0]
    x1 = xy[:, 1]

    # ----- Nonlinear, noisy x2 -----
    # x2 = sin(x0) + 0.5 * x1^2 + noise
    noise_x2 = 0.5 * torch.randn(N, device=device)
    x2 = torch.sin(x0) + 0.5 * x1**2 + noise_x2

    # ----- Heavy-tailed x3 (Student-t like) -----
    # t-dist ~ z / sqrt(u/ν), with z ~ N(0,1), u ~ χ²(ν)
    nu = 3.0
    z = torch.randn(N, device=device)
    u = torch.distributions.Chi2(df=torch.tensor(nu, device=device)).sample((N,))
    x3 = z * torch.sqrt(nu / u)  # Student-t(ν)

    return torch.stack([x0, x1, x2, x3], dim=-1)

Planar flow¶

Lets start with one of the simplest normalizing flow: the planar flow. $$x=f(z)=z+\hat{u}\tanh(w^⊤z+b)$$ where $u$, $w$, and $b$ are learnable parameters.

Note, we use $\hat{u}$, which reparameters $u$ to ensure invertibility. Raw $u$ is unconstrained but $w^T\hat{u} \ge -1$ must hold. So we use the reparameterization $$\hat{u} = u + ({\rm softplus}(w^Tu)-w^Tu-1)\frac{w}{||w||^2}$$

class PlanarFlow(nn.Module):
    def __init__(self, dim):
        super().__init__()
        self.dim = dim

        self.u = nn.Parameter(torch.zeros(dim))
        self.w = nn.Parameter(torch.zeros(dim))
        self.b = nn.Parameter(torch.zeros(())) 

        # initialize
        nn.init.normal_(self.u, mean=0.0, std=0.1)
        nn.init.normal_(self.w, mean=0.0, std=0.1)
        nn.init.zeros_(self.b)

    def _u_hat(self):
        ''' Reparameterization of u to enforce invertibility
        '''
        wtu = torch.dot(self.w, self.u)  # scalar
        # a(x) = softplus(x) - 1
        a = -1.0 + F.softplus(wtu)
        m = -1.0
        w_norm_sq = torch.sum(self.w ** 2)
        u_hat = self.u + (m - a) * self.w / (w_norm_sq + 1e-8)
        return u_hat

    def forward(self, z):
        u_hat = self._u_hat()                    
        w = self.w                                
        b = self.b                          

        linear = z @ w + b
        h = torch.tanh(linear)                

        x = z + h.unsqueeze(1) * u_hat 

        # psi(z) = h'(linear) * w, where h' = 1 - tanh^2
        h_prime = 1.0 - torch.tanh(linear) ** 2  
        psi = h_prime.unsqueeze(1) * w         

        # log |1 + u_hat^T psi|
        u_hat_psi = torch.sum(psi * u_hat, dim=1)  
        log_det = torch.log(torch.abs(1.0 + u_hat_psi) + 1e-8)

        return x, log_det

    def log_prob(self, z):
        _, log_det = self.forward(z)
        return -0.5 * (self.dim * np.log(2 * np.pi) + torch.sum(z ** 2, dim=1)) + log_det

class PlanarFlows(nn.Module):
    def __init__(self, dim, num_flows):
        super().__init__()
        self.dim = dim
        self.flows = nn.ModuleList([PlanarFlow(dim) for _ in range(num_flows)])

    def forward(self, x):
        z = x
        sum_log_det = torch.zeros(z.size(0), device=z.device)

        for flow in self.flows:
            z, log_det = flow(z)
            sum_log_det += log_det

        log_p0 = -0.5 * (self.dim * np.log(2 * np.pi) + torch.sum(z ** 2, dim=1))
        log_px = log_p0 + sum_log_det
        return z, log_px

    def log_prob(self, x):
        _, log_px = self.forward(x)
        return log_px

def create_masks(hidden_dims, device="cpu"):
    # indices for input layer
    input_indices = torch.arange(1, 5, device=device)

    # indices for hidden layers in [1, 3]
    indices = []
    for h in hidden_dims:
        # for each hidden unit, sample an integer index in [1, D-1]
        ind = torch.randint(1, 4, size=(h,), device=device)
        indices.append(ind)

    output_indices = torch.arange(1, 5, device=device).repeat_interleave(2)

    # 4-dimensional input --> mu, \log sigma per dimension
    layer_sizes = [4] + hidden_dims + [8]
    all_indices = [input_indices] + indices + [output_indices]

    masks = []
    for l in range(len(layer_sizes) - 1):
        i_in = all_indices[l]      # [n_in]
        i_out = all_indices[l + 1] # [n_out]
        if l < len(layer_sizes) - 2:
            # hidden layers: allow if d_out >= d_in
            mask = (i_out[:, None] >= i_in[None, :]).float()
        else:
            # output layer: allow if d_out > d_in
            mask = (i_out[:, None] > i_in[None, :]).float()
        masks.append(mask)
    return masks
    
# same MADE class object as the MADE example
class MADE(nn.Module):
    def __init__(self, hidden_dims, D=4, device="cpu"):
        super().__init__()
        self.D = D
        self.hidden_dims = hidden_dims
        self.device = device

        layer_sizes = [D] + hidden_dims + [D * 2]

        layers = []
        for in_f, out_f in zip(layer_sizes[:-1], layer_sizes[1:]):
            layers.append(nn.Linear(in_f, out_f))
        self.layers = nn.ModuleList(layers)

        # build masks
        masks = create_masks(hidden_dims, device=self.device)
        # so they move with .to(device), but are not trainable
        self.masks = nn.ParameterList(
            [nn.Parameter(m, requires_grad=False) for m in masks]
        )
        
        # initialize W and b
        for layer in self.layers:
            nn.init.normal_(layer.weight, mean=0.0, std=1.0 / np.sqrt(layer.in_features))
            nn.init.zeros_(layer.bias)
            
        # near-identity init: last layer outputs ~0
        nn.init.zeros_(self.layers[-1].weight)
        nn.init.zeros_(self.layers[-1].bias)

    def forward(self, x):
        '''forward pass 
        '''
        h = x
        L = len(self.layers)
        for l, layer in enumerate(self.layers):
            # apply autoregressive mask to weights
            masked_weight = layer.weight * self.masks[l]
            h = F.linear(h, masked_weight, layer.bias)
            if l < L - 1:
                h = F.relu(h)
                
        B = h.shape[0]
        h = h.view(B, self.D, 2)
        mean = h[..., 0]
        log_std = h[..., 1]
        return mean, log_std

    def log_prob(self, x):
        ''' log p(x)
        '''
        mean, log_std = self.forward(x)
        inv_std = torch.exp(-log_std)
        z = (x - mean) * inv_std
        log_det = -log_std
        logp_per_dim = -0.5 * (np.log(2.0 * np.pi) + z**2) + log_det
        return torch.sum(logp_per_dim, dim=-1)

class MAFBlock(nn.Module):
    '''single MAF block'''
    def __init__(self, D, hidden_dims, device="cpu"):
        super().__init__()
        self.D = D
        self.net = MADE(hidden_dims, D=D, device=device)

        perm = torch.randperm(D)
        self.register_buffer("perm", perm)
        self.register_buffer("inv_perm", torch.argsort(perm))

    def forward(self, x):
        ''' x -> z 
        '''
        x_perm = x[:, self.perm]

        m, log_s = self.net(x_perm)
        log_s = torch.clamp(log_s, -5, 5)          # prevent exp explosion

        z_perm = (x_perm - m) * torch.exp(-log_s)
        log_det = -log_s.sum(dim=1)                 # sum over D dimensions

        return z_perm[:, self.inv_perm], log_det

    def inverse(self, z):
        ''' z -> x
        '''
        z_perm = z[:, self.perm]
        x_perm = torch.zeros_like(z_perm)

        for i in range(self.D):
            m, log_s = self.net(x_perm)
            log_s = torch.clamp(log_s, -5, 5)
            x_perm[:, i] = z_perm[:, i] * torch.exp(log_s[:, i]) + m[:, i]

        return x_perm[:, self.inv_perm]


class MaskedAutoregressiveFlow(nn.Module):
    ''' MAF with K MAFBlocks and a standard Normal base distribution.
    '''
    def __init__(self, D, hidden_dims, num_flows, device="cpu"):
        super().__init__()
        self.D = D
        self.device = device
        self.flows = nn.ModuleList(
            [MAFBlock(D, hidden_dims, device=device) for _ in range(num_flows)]
        )

    def forward(self, x):
        # x -> z
        z = x
        sum_log_det = torch.zeros(z.size(0), device=z.device)
        for flow in self.flows:
            z, log_det = flow(z)
            sum_log_det += log_det
        return z, sum_log_det

    def log_prob(self, x):
        z, sum_log_det = self.forward(x)
        log_p0 = -0.5 * (self.D * np.log(2 * np.pi) + (z ** 2).sum(dim=1))
        return log_p0 + sum_log_det

    @torch.no_grad()
    def sample(self, num_samples):
        ''' sample z ~ N(0,1) then invert'''
        self.eval()
        z = torch.randn(num_samples, self.D, device=self.device)
        for flow in reversed(self.flows):           # invert in reverse order
            z = flow.inverse(z)
        return z
