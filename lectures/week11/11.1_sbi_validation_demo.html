def simulation(theta): 
    a, b = theta
    noise = yerr * np.random.normal(size=len(x))
    return a + b * x + noise

def sample_prior(): 
    # sample a and b from uniform prior
    _a = -0.5 + np.random.normal() 
    _b = 5 + 3*np.random.normal()
    return [_a, _b]

posterior predictive test¶

One common validation step is the posterior predictive check

$$\theta' \sim q_\phi(\theta|\mathcal{D}_{\rm obs})$$ then compare $F(\theta')$ to $\mathcal{D}_{\rm obs}$

simulation-based calibration (SBC)¶

Let's start with SBC

The algorithm is simple:

for theta_test, data_test in test_samples: 

    # draw samples from posterior estimate
    q_samples = NPE(data_test)

    # compute rank 
    rank = sum(q_samples < theta_test)
In [14]:
ranks_a = []
ranks_b = [] 

for _theta, _data in zip(theta_test, data_test): 
    _q_samples = npe_maf.sample((2000,), 
                                x=torch.tensor(_data, dtype=torch.float32), 
                                show_progress_bars=False)

    # calculate rank for each dimension
    ranks_a.append(np.sum(_q_samples.numpy()[:,0] < _theta[0])) 
    ranks_b.append(np.sum(_q_samples.numpy()[:,1] < _theta[1]))
